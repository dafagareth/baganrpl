\newcommand{\drawactor}[4]{%
  \coordinate (#3) at (#1,#2);
  \begin{scope}[shift={(#1,#2)},thick]
    \draw (0,0.5) circle (0.22);
    \draw (0,0.28) -- (0,-0.28);
    \draw (-0.34,0.08) -- (0.34,0.08);
    \draw (0,-0.28) -- (-0.27,-0.74);
    \draw (0,-0.28) -- (0.27,-0.74);
  \end{scope}
  \node[font=\small\bfseries, align=center] at (#1,#2-1.18) {#4};
}
